const CF_WORKER = 'https://macrologix-tms-api.vince-b45.workers.dev';

// ═══════════════════════════════════════════════════════════════
// SUPABASE HELPERS
// ═══════════════════════════════════════════════════════════════
async function dbGet(table, query) {
  try {
    const url = CF_WORKER + '/api/' + table + (query ? '?' + query : '');
    const r = await fetch(url);
    return r.ok ? await r.json() : [];
  } catch(e) { console.error('GET', e); return []; }
}

async function dbPost(table, body) {
  try {
    const r = await fetch(CF_WORKER + '/api/' + table, {
      method: 'POST', headers: {'Content-Type':'application/json'},
      body: JSON.stringify(body)
    });
    return r.ok ? (await r.json().catch(()=>({}))) : null;
  } catch(e) { console.error('POST', e); return null; }
}

async function dbPatch(table, col, val, body) {
  try {
    const r = await fetch(CF_WORKER + '/api/' + table + '?' + col + '=eq.' + encodeURIComponent(val), {
      method: 'PATCH', headers: {'Content-Type':'application/json'},
      body: JSON.stringify(body)
    });
    return r.ok;
  } catch(e) { console.error('PATCH', e); return false; }
}

async async function doLogin() {
  const username = (document.getElementById('l-username').value || '').trim().toLowerCase();
  const password = (document.getElementById('l-password').value || '');
  const errEl    = document.getElementById('login-err');
  const btn      = document.getElementById('login-btn');
  errEl.style.display = 'none';
  if (!username || !password) { errEl.textContent = 'Enter username and password.'; errEl.style.display = 'block'; return; }

  btn.textContent = 'Signing in…'; btn.disabled = true;

  try {
    // Look up user by username
    const rows = await dbGet('ml_customer_users', 'username=eq.' + encodeURIComponent(username));
    const user = rows && rows[0];
    if (!user || user.password_hash !== password) {
      errEl.textContent = 'Invalid username or password.';
      errEl.style.display = 'block';
      btn.textContent = 'Sign In'; btn.disabled = false;
      return;
    }
    // Get customer info
    const custRows = await dbGet('ml_customers', 'id=eq.' + user.customer_id);
    const cust = custRows && custRows[0];
    const session = {
      userId:       user.id,
      customerId:   user.customer_id,
      customerName: cust ? cust.name : username,
      username:     user.username,
      role:         user.role || 'viewer'
    };
    setSession(session);
    _session = session;
    // Update last login
    dbPatch('ml_customer_users', 'id', user.id, { last_login: new Date().toISOString() });
    bootApp();
  } catch(e) {
    errEl.textContent = 'Error connecting. Please try again.';
    errEl.style.display = 'block';
    btn.textContent = 'Sign In'; btn.disabled = false;
  }
}

  clearSession();
  document.getElementById('app').style.display = 'none';
  document.getElementById('login-screen').style.display = 'flex';
  document.getElementById('l-password').value = '';
}

// ═══════════════════════════════════════════════════════════════
// BOOT
// ═══════════════════════════════════════════════════════════════
async async function bootApp() {
  const s = _session || getSession();
  if (!s) return;
  _session = s;

  // If customerName is missing (came from landing page sign-in), fetch it from DB
  if (!s.customerName && s.customerId) {
    const custRows = await dbGet('ml_customers', 'id=eq.' + s.customerId + '&select=name,payment_terms,billing_email,status');
    const cust = custRows && custRows[0];
    if (cust) {
      s.customerName  = cust.name          || 'Customer';
      s.paymentTerms  = cust.payment_terms || '';
      s.billingEmail  = cust.billing_email || '';
      s.status        = cust.status        || 'active';
      setSession(s); // persist updated session
    }
  }

  // Update last login timestamp
  if (s.userId) {
    dbPatch('ml_customer_users', 'id', s.userId, { last_login: new Date().toISOString() });
  }

  // Update header
  const initials = (s.name||'?').split(' ').map(w=>w[0]||'').join('').toUpperCase().slice(0,2);
  document.getElementById('hdr-avatar').textContent  = initials;
  document.getElementById('hdr-name').textContent    = s.name;
  document.getElementById('hdr-company').textContent = s.customerName || '';

  document.getElementById('login-screen').style.display = 'none';
  document.getElementById('app').style.display = 'flex';
  document.getElementById('app').style.flexDirection = 'column';

  // Update welcome
  const hour = new Date().getHours();
  const greet = hour < 12 ? 'Good morning' : hour < 17 ? 'Good afternoon' : 'Good evening';
  const firstName = (s.name||'').split(' ')[0] || 'there';
  const dashWelcome = document.getElementById('dash-welcome');
  if (dashWelcome) dashWelcome.textContent = greet + ', ' + firstName + '!';
  const dashSub = document.getElementById('dash-sub');
  if (dashSub) dashSub.textContent = (s.customerName||'Your account') + ' — Here\'s a summary of your current shipments and account status.';

  // Populate account page
  const sv = (id,v)=>{ const el=document.getElementById(id); if(el) el.value=v||''; };
  sv('acc-name', s.name); sv('acc-username', s.username); sv('acc-email', s.email);
  sv('acc-role', s.role); sv('acc-company', s.customerName);
  sv('acc-billing-email', s.billingEmail); sv('acc-terms', s.paymentTerms); sv('acc-status', s.status);

  // Load all data in parallel
  await Promise.all([ loadLoads(), loadInvoices(), loadTrailers() ]);

  renderDashboard();
}

// ═══════════════════════════════════════════════════════════════
// DATA LOADERS
// ═══════════════════════════════════════════════════════════════
async function loadLoads() {
  const cid   = _session.customerId;
  const cname = _session.customerName;
  // Fetch by both customer_id and customer name in parallel, merge and dedup
  const [byId, byName] = await Promise.all([
    dbGet('ml_loads', 'customer_id=eq.' + cid + '&order=created_at.desc&limit=200'),
    dbGet('ml_loads', 'customer=eq.' + encodeURIComponent(cname) + '&order=created_at.desc&limit=200')
  ]);
  const seen = new Set();
  const merged = [];
  for (const row of [...(byId||[]), ...(byName||[])]) {
    if (!seen.has(row.id)) { seen.add(row.id); merged.push(row); }
  }
  // Sort by created_at desc
  merged.sort((a,b) => new Date(b.created_at||0) - new Date(a.created_at||0));
  _loads = merged;
  renderLoadsTable();
}

async function loadInvoices() {
  const cname = _session.customerName;
  let rows = await dbGet('ml_invoices',
    'customer=eq.' + encodeURIComponent(cname) + '&order=created_at.desc&limit=100');
  _invoices = rows || [];
  renderInvoicesTable();
}

async function loadTrailers() {
  const cid = _session.customerId;
  // Fetch trailers where customer_id matches AND dropped/available status
  let rows = await dbGet('ml_trailers',
    'customer_id=eq.' + cid + '&order=id.asc');
  _trailers = rows || [];
  renderTrailersTable();
}

// ═══════════════════════════════════════════════════════════════
// INVOICES TABLE
// ═══════════════════════════════════════════════════════════════
function renderInvoicesTable() {
  const tbody = document.getElementById('invoices-tbody');
  if (!tbody) return;
  const now = new Date();
  let outstanding = 0, overdue = 0, paidYTD = 0;

  if (!_invoices.length) {
    tbody.innerHTML = '<tr><td colspan="7" style="text-align:center;color:var(--text3);padding:32px">No invoices found</td></tr>';
  } else {
    tbody.innerHTML = _invoices.map(inv => {
      const isOver = inv.due_date && inv.status!=='paid' && new Date(inv.due_date) < now;
      const st = isOver ? 'overdue' : inv.status;
      if (st !== 'paid' && st !== 'cancelled') outstanding += Number(inv.amount||0);
      if (isOver) overdue += Number(inv.amount||0);
      if (inv.status==='paid' && inv.paid_at && new Date(inv.paid_at).getFullYear()===now.getFullYear()) paidYTD += Number(inv.amount||0);
      const issued = inv.issued_date ? new Date(inv.issued_date).toLocaleDateString('en-US',{month:'short',day:'numeric',year:'numeric'}) : '—';
      const due    = inv.due_date    ? new Date(inv.due_date).toLocaleDateString('en-US',{month:'short',day:'numeric',year:'numeric'}) : '—';
      return `<tr>
        <td class="mono color-accent fw-700" style="cursor:pointer" onclick="openInvoiceModal('${inv.id}')">${inv.id}</td>
        <td class="mono" style="font-size:11px">${(inv.load_ids||[]).join(', ')||'—'}</td>
        <td class="fw-700 color-green">$${Number(inv.amount||0).toLocaleString(undefined,{minimumFractionDigits:2})}</td>
        <td style="font-size:12px">${issued}</td>
        <td style="font-size:12px${isOver?';color:var(--red)':''}">${due}</td>
        <td>${invBadge(st)}</td>
        <td class="table-actions">
          <button class="tbl-btn" onclick="openInvoiceModal('${inv.id}')">View</button>
          <button class="tbl-btn" onclick="printInvoice('${inv.id}')">🖨 Print</button>
          <button class="tbl-btn" onclick="emailInvoice('${inv.id}')">✉️ Email</button>
        </td>
      </tr>`;
    }).join('');
  }

  const f = (n) => '$' + n.toLocaleString(undefined,{minimumFractionDigits:2});
  const sk = (id,v) => { const el=document.getElementById(id); if(el) el.textContent=v; };
  sk('inv-outstanding', f(outstanding));
  sk('inv-overdue',     f(overdue));
  sk('inv-paid-ytd',   f(paidYTD));
}

// TRAILERS TABLE
// ═══════════════════════════════════════════════════════════════
function renderTrailersTable() {
  const tbody = document.getElementById('trailers-tbody');

  const sumEl = document.getElementById('trailers-summary');
  if (!_trailers.length) {
    tbody.innerHTML = '<tr><td colspan="7" style="text-align:center;color:var(--text3);padding:32px">No trailers currently at your facility</td></tr>';
    if (sumEl) sumEl.innerHTML = '';
    return;
  }

  const statusColors = { available:'badge-green', dropped:'badge-amber', in_use:'badge-blue',
                         Dropped:'badge-amber', Available:'badge-green', 'On Truck':'badge-blue' };
  tbody.innerHTML = _trailers.map(t => {
    const upd = t.last_update||t.last_event_at||t.updated_at;
    const updFmt = upd ? new Date(upd).toLocaleDateString('en-US',{month:'short',day:'numeric',hour:'2-digit',minute:'2-digit'}) : '—';
    return `<tr>
      <td class="mono fw-700 color-accent">${t.id}</td>
      <td>${t.trailer_type||t.type||'—'}</td>
      <td><span class="badge ${statusColors[t.status]||'badge-gray'}">${t.status||'—'}</span></td>
      <td style="font-size:12px">${t.drop_location||t.location||t.yard||'—'}</td>
      <td style="font-size:12px">${t.driver_id||'—'}</td>
      <td class="mono" style="font-size:11px;color:var(--accent)">${t.load_id||'—'}</td>
      <td style="font-size:11px;color:var(--text3)">${updFmt}</td>
    </tr>`;
  }).join('');

  if (sumEl) {
    const dropped  = _trailers.filter(t=>t.status==='dropped'||t.status==='Dropped').length;
    const avail    = _trailers.filter(t=>t.status==='available'||t.status==='Available').length;
    sumEl.innerHTML =
      `<span class="badge badge-gray">${_trailers.length} Total</span>` +
      (dropped ? `<span class="badge badge-amber">📍 ${dropped} Dropped</span>` : '') +
      (avail   ? `<span class="badge badge-green">✅ ${avail} Available</span>` : '');

function orderNext(step) {
  // Validate
  const required = {
    1: ['o-pu-name','o-pu-addr','o-pu-city','o-pu-state','o-pu-date'],
    2: ['o-del-name','o-del-addr','o-del-city','o-del-state','o-del-date'],
    3: ['o-commodity','o-weight']
  };
  if (required[step]) {
    for (const id of required[step]) {
      const el = document.getElementById(id);
      if (!el || !el.value.trim()) {
        el && el.focus();
        toast('Please fill in all required fields (*)');
        return;
      }
    }
  }

  document.getElementById(`order-step-${step}`).style.display = 'none';
  document.getElementById(`order-step-${step+1}`).style.display = '';
  updateWizardSteps(step+1);

  if (step === 3) buildOrderReview();
}
function orderBack(step) {
  document.getElementById(`order-step-${step}`).style.display = 'none';
  document.getElementById(`order-step-${step-1}`).style.display = '';
  updateWizardSteps(step-1);
}
function updateWizardSteps(active) {
  for (let i=1;i<=4;i++) {
    const el = document.getElementById(`ws-${i}`);
    if (!el) continue;
    el.classList.remove('active','done');
    if (i < active)  el.classList.add('done');
    if (i === active) el.classList.add('active');
    if (i < 4) {
      const line = document.getElementById(`sl-${i}`);
      if (line) line.style.background = i < active ? 'var(--accent)' : '';
    }
  }
}

      notification_sent:  true,

      updated_at:         now

async function changePassword() {
  const cur  = document.getElementById('acc-cur-pass').value;
  const nw   = document.getElementById('acc-new-pass').value;
  const conf = document.getElementById('acc-conf-pass').value;
  if (!cur || !nw) { toast('Enter current and new password'); return; }
  if (nw !== conf) { toast('New passwords do not match'); return; }
  if (nw.length < 6) { toast('Password must be at least 6 characters'); return; }

  // Verify current against DB
  const rows = await dbGet('ml_customer_users', 'id=eq.' + _session.userId);
  if (!rows || !rows[0] || rows[0].password_hash !== cur) {
    toast('Current password is incorrect'); return;
  }
  await dbPatch('ml_customer_users', 'id', _session.userId, {
    password_hash: nw, updated_at: new Date().toISOString()
  });
  document.getElementById('acc-cur-pass').value = '';
  document.getElementById('acc-new-pass').value = '';
  document.getElementById('acc-conf-pass').value = '';
  toast('✓ Password updated successfully');
}

// NAVIGATION
// ═══════════════════════════════════════════════════════════════
function showPage(id, el) {
  ['dashboard','loads','invoices','trailers','order','account'].forEach(p => {
    const pageEl = document.getElementById('page-' + p);
    if (pageEl) pageEl.style.display = p === id ? '' : 'none';
  });
  document.querySelectorAll('.cp-nav-item').forEach(n => n.classList.remove('active'));
  if (el) el.classList.add('active');
  else {
    const navMap = {dashboard:0,loads:1,invoices:2,trailers:3,order:4,account:5};
    const idx = navMap[id];
    const items = document.querySelectorAll('.cp-nav-item');
    if (idx !== undefined && items[idx]) items[idx].classList.add('active');
  }

  if (id === 'trailers')  { loadTrailers().then(renderTrailersTable); }
  if (id === 'loads')     { loadLoads().then(()=>renderLoadsTable()); }
  if (id === 'invoices')  { loadInvoices().then(renderInvoicesTable); }
  if (id === 'order' && document.getElementById('order-step-1')) {
    // Reset wizard to step 1
    for (let i=1;i<=4;i++) {
      const s = document.getElementById('order-step-' + i);
      if (s) s.style.display = i===1 ? '' : 'none';
    }
    updateWizardSteps(1);
  }
}

function fmt(dt) {
  try { return new Date(dt).toLocaleDateString('en-US',{month:'short',day:'numeric',year:'numeric'}); }
  catch(e) { return dt||'—'; }
}
function toast(msg) {
  const el = document.getElementById('toast');
  el.textContent = msg; el.style.display = 'block';
  clearTimeout(toast._t);
  toast._t = setTimeout(() => el.style.display = 'none', 3000);
}
function statusBadge(s) {
  const map = { open:'badge-gray', dispatched:'badge-blue', intransit:'badge-green', delayed:'badge-amber', delivered:'badge-purple', invoiced:'badge-green' };
  const lbl = { open:'Open', dispatched:'Dispatched', intransit:'In Transit', delayed:'Delayed', delivered:'Delivered', invoiced:'Invoiced' };
  return `<span class="badge ${map[s]||'badge-gray'}">${lbl[s]||s||'—'}</span>`;
}

}
